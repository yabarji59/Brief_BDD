- Créer un user sur SQL Shell
CREATE ROLE Leo;

- Attribuer le rôle "création de base de données"
ALTER ROLE Leo CREATEDB;

- Attribuer le rôle "superuser"
ALTER ROLE Leo SUPERUSER;

- Changer le nom d'un user
ALTER ROLE Leo RENAME TO Lea;

- Créer une database en specifiant le owner
CREATE DATABASE Library OWNER Lea;

- Créer une table
CREATE TABLE leads (id INTEGER PRIMARY KEY, name VARCHAR);

- Modifier un champs dans une table
UPDATE membres SET prenom = 'Ron'
	WHERE membres.num_carte = 1031980;

- Supprimer un champs dans une table
DELETE FROM membres
	WHERE membres.num_carte = 1031980;

- Effectuer une recherche à partir d'un nom
SELECT * FROM membres
	WHERE nom = 'Potter';

- Effectuer une recherche entre deux dates
SELECT * FROM membres
	WHERE birthday > '1975-01-01' AND birthday < '1985-01-01';
